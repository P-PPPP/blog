Step2 编译#

编译 Clang5.05#

#下载llvm源码
wget http://llvm.org/releases/5.0.2/llvm-5.0.2.src.tar.xz
#解压llvm
tar xf llvm-5.0.2.src.tar.xz && mv llvm-5.0.2.src llvm
#下载Clang源码
cd llvm/tools
wget http://llvm.org/releases/5.0.2/cfe-5.0.2.src.tar.xz
tar xf cfe-5.0.2.src.tar.xz && mv cfe-5.0.2.src clang
#下载Clang-tools-extra源码
cd clang/tools
wget http://llvm.org/releases/5.0.2/clang-tools-extra-5.0.2.src.tar.xz
tar xf clang-tools-extra-5.0.2.src.tar.xz && mv clang-tools-extra-5.0.2.src  extra
#下载compiler-rt源码
cd llvm/projects
wget http://llvm.org/releases/5.0.2/compiler-rt-5.0.2.src.tar.xz
tar xf compiler-rt-5.0.2.src.tar.xz && mv compiler-rt-5.0.2.src compiler-rt
#编译

mkdir build && cd build

export CC=/usr/bin/gcc
export CXX=/usr/bin/g++
export installPath=/usr/bin/clang

cmake -G "Unix Makefiles" --enable-targets=aarch64 -DCMAKE_BUILD_TYPE=Release ../ -DCMAKE_INSTALL_PREFIX=$installPath

（cmake 这一步也可以指定安装路径，添加-DCMAKE_INSTALL_PREFIX=$installPath，最终将安装到$installPath，否则默认安装到/usr/local）

make -j $(nproc)

make install

Qemu 编译#

#Qemu编译
cd qemu 
./configure
make -j$(nproc)
sudo make install
